Compose and render a multi-track video using the Rendley engine. Supports clips, text, audio, transitions, effects, filters, and keyframe animations.

<Info>
**Timeout:** 600s | **Provider:** triggerdev | **Accepts:** video
</Info>

## Request

<CodeGroup>

```bash cURL
curl -X POST https://api.rendobar.com/jobs \
  -H "Authorization: Bearer rb_your_key" \
  -H "Content-Type: application/json" \
  -d '{
  "type": "render.rendley",
  "inputs": {},
  "params": {
    "width": 1920,
    "height": 1080,
    "clips": [
      {
        "type": "video",
        "url": "https://cdn.rendobar.com/assets/examples/sample.mp4",
        "startTime": 0,
        "duration": 10
      }
    ]
  }
}'
```

```javascript JavaScript
const response = await fetch("https://api.rendobar.com/jobs", {

```

```python Python
import requests

response = requests.post(
```

## Parameters

<ParamField body="width" type="integer" default="1920">
  Canvas width in pixels
</ParamField>

<ParamField body="height" type="integer" default="1080">
  Canvas height in pixels
</ParamField>

<ParamField body="backgroundColor" type="string" default="#000000">
  Background color in hex format
</ParamField>

<ParamField body="clips" type="object[]" required>
  Array of clips to compose on the timeline
  <Expandable title="item properties">
    <ParamField body="type" type="video | image | audio | text | lottie" required>
      Clip media type
    </ParamField>

    <ParamField body="url" type="string">
      URL to the media file
    </ParamField>

    <ParamField body="startTime" type="number" default="0">
      Position on timeline in seconds
    </ParamField>

    <ParamField body="duration" type="number">
      Clip duration in seconds
    </ParamField>

    <ParamField body="trimStart" type="number">
      Trim from source start in seconds
    </ParamField>

    <ParamField body="trimEnd" type="number">
      Trim from source end in seconds
    </ParamField>

    <ParamField body="layer" type="integer" default="0">
      Layer index (0 = bottom)
    </ParamField>

    <ParamField body="style" type="object">
      Clip position, scale, rotation, and opacity
      <Expandable title="properties">
        <ParamField body="x" type="number" default="0">
          X position on canvas
        </ParamField>

        <ParamField body="y" type="number" default="0">
          Y position on canvas
        </ParamField>

        <ParamField body="width" type="number">
          Width in pixels
        </ParamField>

        <ParamField body="height" type="number">
          Height in pixels
        </ParamField>

        <ParamField body="rotation" type="number" default="0">
          Rotation angle in degrees
        </ParamField>

        <ParamField body="opacity" type="number" default="1">
          Opacity (0 = transparent, 1 = opaque)
        </ParamField>

      </Expandable>
    </ParamField>

    <ParamField body="text" type="object">
      Text clip properties (required when type is text)
      <Expandable title="properties">
        <ParamField body="content" type="string" required>
          Text content to display
        </ParamField>

        <ParamField body="fontFamily" type="string" default="Arial">
          Font family name
        </ParamField>

        <ParamField body="fontSize" type="integer" default="48">
          Font size in pixels
        </ParamField>

        <ParamField body="color" type="string" default="#FFFFFF">
          Text color in hex format
        </ParamField>

      </Expandable>
    </ParamField>

    <ParamField body="filters" type="string[]">
      Filter IDs to apply to this clip
    </ParamField>

    <ParamField body="effects" type="string[]">
      Effect IDs to apply to this clip
    </ParamField>

  </Expandable>
</ParamField>

<ParamField body="transitions" type="object[]">
  Transitions between clips
  <Expandable title="item properties">
    <ParamField body="type" type="string" required>
      Transition type name
    </ParamField>

    <ParamField body="duration" type="number" required>
      Transition duration in seconds
    </ParamField>

    <ParamField body="fromClipIndex" type="integer" required>
      Index of the outgoing clip
    </ParamField>

    <ParamField body="toClipIndex" type="integer" required>
      Index of the incoming clip
    </ParamField>

  </Expandable>
</ParamField>

<ParamField body="fonts" type="object[]">
  Custom fonts to load for text clips
  <Expandable title="item properties">
    <ParamField body="family" type="string" required>
      Font family name
    </ParamField>

    <ParamField body="url" type="string" required>
      URL to the font file
    </ParamField>

    <ParamField body="weight" type="number">
      Font weight (e.g. 400, 700)
    </ParamField>

    <ParamField body="style" type="string">
      Font style (e.g. normal, italic)
    </ParamField>

  </Expandable>
</ParamField>

<ParamField body="subtitles" type="object[]">
  Subtitles to overlay on the composition
  <Expandable title="item properties">
    <ParamField body="url" type="string" required>
      URL to the subtitle file
    </ParamField>

    <ParamField body="format" type="srt | vtt | ass" default="srt">
      Subtitle file format
    </ParamField>

  </Expandable>
</ParamField>
